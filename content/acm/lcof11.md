#include <bits/stdc++.h>
using namespace std;

class Solution {
private:
    int isSmaller(vector<int>& arr,int n){
        int l = 0,r = n - 1;
        while(l < r){
            int mid = r + l >> 1;
            if(arr[mid] > arr[r])
                l = mid + 1;
            else if(arr[l] == arr[r]){
                l ++;
                r --;
            }
            else
                r = mid;
        }
        return l;
    }
public:
    int minArray(vector<int>& numbers) {
        int ind = isSmaller(numbers,numbers.size());
        return numbers[ind];
    }
};
