 #include <leetcode.h>

class Solution {
public:
    ListNode* removeNthFromEnd(ListNode* head, int n) {
        ListNode* slow=head,*fast=head,*last=new ListNode(0,head);
        ListNode* flag=last;
        for(int i=1;i<=n-1&&fast->next;i++)
            fast=fast->next;
        while(fast->next!=nullptr){
            fast=fast->next;
            last=slow;
            slow=slow->next;
        }
        // cout<<slow->val<<endl;
        last->next=slow->next;
        return flag->next;
    }
};
